Laravel 使用JWT API 授权🚀

什么是JWT

Json web token (JWT), 是为了在网络应用环境间传递声明而执行的一种基于JSON的开放标准（(RFC 7519).该token被设计为紧凑且安全的，特别适用于分布式站点的单点登录（SSO）场景。JWT的声明一般被用来在身份提供者和服务提供者间传递被认证的用户身份信息，以便于从资源服务器获取资源，也可以增加一些额外的其它业务逻辑所必须的声明信息，该token也可直接被用于认证，也可被加密。作者：Dearmadman链接：https://www.jianshu.com/p/576dbf44b2ae来源：简书著作权归作者所有。商业转载请联系作者获得授权，非商业转载请注明出处。

关于JWT讲解与分析：https://learnku.com/articles/17883

安装

composer 安装 lcobucci/jwt ，我这里默认安装的是最新版 4.1 。

composer require lcobucci/jwt

封装

在 app\Helpers 目录下新建 Jwt.php 文件，封装 Jwt 类。

<?php
/**
 * Created by PhpStorm.
 * User: zhukang
 * Date: 2019/8/22
 * Time: 16:39
 */

namespace App\Helpers;

use Illuminate\Support\Facades\Redis;
use Lcobucci\Clock\SystemClock;
use Lcobucci\JWT\Configuration;
use Lcobucci\JWT\Exception;
use Lcobucci\JWT\Signer\Hmac\Sha256;
use Lcobucci\JWT\Signer\Key\InMemory;
use Lcobucci\JWT\Token\Plain;
use Lcobucci\JWT\Validation\Constraint\IdentifiedBy;
use Lcobucci\JWT\Validation\Constraint\IssuedBy;
use Lcobucci\JWT\Validation\Constraint\PermittedFor;
use Lcobucci\JWT\Validation\Constraint\SignedWith;
use Lcobucci\JWT\Validation\Constraint\ValidAt;
use Lcobucci\JWT\Validation\RequiredConstraintsViolated;

trait Jwt
{
    protected $issuedBy;//签发人
    protected $permittedFor;//受众
    protected $identifiedBy;//jwt编号
    protected $issuedAt;//签发时间
    protected $expiresAt;//过期时间
    protected $jwt_secret; //秘钥

    public function __construct()
    {
        //获取jwt配置信息
        $jwt_configs        = config('jwt');
        $this->issuedBy     = $jwt_configs['issuedBy'];
        $this->permittedFor = $jwt_configs['permittedFor'];
        $this->identifiedBy = $jwt_configs['identifiedBy'];
        $this->issuedAt     = $jwt_configs['issuedAt'];
        $this->expiresAt    = $this->issuedAt->modify($jwt_configs['expiresAt']);
        $this->jwt_secret   = $jwt_configs['jwt_secret'];
    }

    /**
     * @Description:生成jwt配置对象
     * @return Configuration
     * @Author:zhukang
     * @Date:2022/4/7 16:05
     */
    public function createJwtObject()
    {
        return Configuration::forSymmetricSigner(
            new Sha256(),
            InMemory::base64Encoded($this->jwt_secret)
        );
    }

    /**
     * @Description:生成token
     * @param array $bind 需要在jwt载体(payload)中加入的数据  ['userid'=>1,'xxxx'=>xxxx]
     * @return string \Lcobucci\JWT\Token
     * @Author:zhukang
     * @Date:2022/4/7 16:04
     */
    public function getToken(array $bind = [])
    {
        $jwt_object = $this->createJwtObject();
        $jwt_obj    = $jwt_object->builder()
            ->issuedBy($this->issuedBy)
            ->permittedFor($this->permittedFor)
            ->identifiedBy($this->identifiedBy)
            ->issuedAt($this->issuedAt)
            ->canOnlyBeUsedAfter($this->issuedAt)
            ->expiresAt($this->expiresAt);
        if (is_array($bind) && !empty($bind)) {
            foreach ($bind as $k => $v) {
                $jwt_obj->withClaim($k, $v);
            }
        }
        //签名算法和签名秘钥得到token
        return $jwt_obj->getToken($jwt_object->signer(), $jwt_object->signingKey())->toString();
    }

    /**
     * @Description:验证token
     * @param $token
     * @return bool
     * @Author:zhukang
     * @Date:2022/4/7 16:04
     */
    public function verifyToken($token)
    {
        $jwt_object = $this->createJwtObject();
        try {
            $token = $jwt_object->parser()->parse($token);
        } catch (\Exception $e) {
            return false;
        }
        $validate_jwt_id = new IdentifiedBy($this->identifiedBy);
        $jwt_object->setValidationConstraints($validate_jwt_id);
        $validate_issued = new IssuedBy($this->issuedBy);
        $jwt_object->setValidationConstraints($validate_issued);
        $validate_aud = new PermittedFor($this->permittedFor);
        $jwt_object->setValidationConstraints($validate_aud);
        //验证是否过期
        $timezone        = new \DateTimeZone('PRC');
        $time            = new SystemClock($timezone);
        $validate_jwt_at = new ValidAt($time);
        $jwt_object->setValidationConstraints($validate_jwt_at);
        $validate_contents = $jwt_object->validationConstraints();
        try {
            $jwt_object->validator()->assert($token, ...$validate_contents);
            return true;
        } catch (RequiredConstraintsViolated $e) {
            return false;
        }
    }

    /**
     * @Description:解析token里面的载体数据信息
     * @param $token
     * @return mixed
     * @Author:zhukang
     * @Date:2022/4/7 16:03
     */
    public function getTokenContent($token)
    {
        $jwt_object = $this->createJwtObject();
        try {
            $token  = $jwt_object->parser()->parse($token);
            $claims = json_decode(base64_decode($token->claims()->toString()), true);
        } catch (\Exception $e) {
            return $this->fail([405001, '非法token值！']);
        }
        return $claims;
    }

    /**
     * @Description:验证token是否在刷新有效期内
     * @param $token
     * @return bool
     * @Author:zhukang
     * @Date:2022/4/7 16:03
     */
    public function verifyRefresh($token)
    {
        $jwt_object = $this->createJwtObject();
        try {
            $token = $jwt_object->parser()->parse($token);
        } catch (\Exception $e) {
            return false;
        }

        $claims = json_decode(base64_decode($token->claims()->toString()), true);

        // 签发时间大于当前服务器时间验证失败
        if (isset($claims['iat']) && $claims['iat'] > time()) {
            return $claims;
        }

        // 生效时间大于当前服务器时间验证失败
        if (isset($claims['nbf']) && $claims['nbf'] > time()) {
            return false;
        }

        // 当前服务器时间大于过期时间验证失败
        if (isset($claims['exp']) && time() > $claims['exp']) {
            return false;
        }

        return true;
    }

    /**
     * @Description:验证签名
     * @param $token
     * @return bool
     * @Author:zhukang
     * @Date:2022/4/7 16:17
     */
    public function verifySign($token)
    {
        $jwt_object = $this->createJwtObject();
        try {
            $token = $jwt_object->parser()->parse($token);
        } catch (\Exception $e) {
            return false;
        }

        $validate_jwt_signer = new SignedWith($jwt_object->signer(), $jwt_object->signingKey());
        $jwt_object->setValidationConstraints($validate_jwt_signer);
        $validate_contents = $jwt_object->validationConstraints();
        try {
            $jwt_object->validator()->assert($token, ...$validate_contents);
            return true;
        } catch (RequiredConstraintsViolated $e) {
            return $this->fail([405001, $e->getMessage()]);
        }
    }

    /**
     * @Description:使Token失效
     * @return string
     * @Author:zhukang
     * @Date:2022/4/7 16:18
     */
    public function invalidate()
    {
        return '';
    }

    /**
     * @Description:将token加入黑名单
     * @param $token
     * @return mixed
     * @Author:zhukang
     * @Date:2022/4/7 16:19
     */
    public function addBlacklist($token)
    {
        $token_redis_ex = $this->getTokenContent($token)['exp'] - time();

        // 设置成功返回true 失败false
        $result = Redis::setex($token, $token_redis_ex, $token);

        return $result;
    }
}

配置

在 config 目录下，新建 Jwt配置文件。

<?php

return [

    /*
    |--------------------------------------------------------------------------
    | Jwt Configuration
    |--------------------------------------------------------------------------
    |
    | lcobucci/jwt is a framework-agnostic PHP library that allows you to
    | issue, parse, and validate JSON Web Tokens based on the RFC 7519.
    |
    */

    //签发人
    'issuedBy'     => env('APP_NAME'),

    //受众
    'permittedFor' => 'client-user',

    //jwt编号
    'identifiedBy' => 'admin',

    //签发时间
    'issuedAt'     => new DateTimeImmutable(),

    //过期时间
    'expiresAt'    => '+1 hour',

    //秘钥

使用

在 app\Http\Middleware 目录下，新建 AuthApi.php 文件。

<?php

namespace App\Http\Middleware;

use App\Helpers\Jwt;
use App\Models\User;
use Closure;
use App\Helpers\ApiResponse;
use Illuminate\Support\Facades\Redis;

class AuthApi
{
    use Jwt, ApiResponse;

    /**
     * @Description:Handle an incoming request.
     * @param $request
     * @param Closure $next
     * @return \Illuminate\Http\JsonResponse|mixed
     * @Author:zhukang
     * @Date:2022/4/8 10:53
     */
    public function handle($request, Closure $next)
    {
        //获取token
        $token = $request->bearerToken();

        //判断token知否为空
        if (empty($token)) {
            return $this->fail([4001, 'Token is missing!']);
        }

        //判断redis是否还存在该token
        if (Redis::get($token) != '') {
            return $this->fail([4002, 'Login authorization has expired!']);
        }

        //验证token签名是否合法
        if (is_array($this->verifySign($token)) || $this->verifySign($token) !== true) {
            return $this->fail([4004, 'The token is invalid, please replace it.']);
        }

        //判断该token所绑定的用户ID是否真实存在
        $info = $this->getTokenContent($token);
        $user = User::where('id', $info['user_id'])->first();
        if (empty($user)) {
            return $this->fail([4004, 'The token is invalid, please replace it.']);
        }

        // 验证token是否有效
        $verify_res = $this->verifyToken($token);
        if (!$verify_res) {
            // 验证token是否在刷新有效期内
            $legal = $this->verifyRefresh($token);
            if (!$legal) {
                return $this->fail([4003, 'Login authorization has expired!']);
            }

            $user_data = $this->getTokenContent($token);// 获取原token中的数据
            $token     = $this->getToken($user_data);// 重新生成token
            header('token:' . $token);// 设置响应头

            return $this->fail([4004, 'The token is invalid, please replace it.']);
        }

        //判断账户状态
        if (empty($user) || $user->status != 1) {
            return $this->fail([4044, 'Account anomaly, please contact CS!']);
        }

        return $next($request);
    }
}
