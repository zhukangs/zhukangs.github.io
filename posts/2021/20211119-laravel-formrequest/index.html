新建场景验证类

  • 新建 app/Http/Requests/SceneValidator.php
<?php
    
namespace App\Http\Requests;

use Illuminate\Contracts\Validation\{Factory, Validator};

trait SceneValidator
{
    protected $scene = null;

    protected $onlyRule = [];

    protected $autoValidate = true;

    /**
     * Validate.
     *
     * @param string|array $scene
     */
    public function validate($scene = '')
    {
        if (!$this->autoValidate) {
            if (is_array($scene)) {
                $this->onlyRule = $scene;
            } else {
                $this->scene = $scene;
            }

            $this->handleValidate();
        }
    }

    /**
     * 覆盖 ValidatesWhenResolvedTrait->validateResolved
     */
    public function validateResolved()
    {
        if (method_exists($this, 'autoValidate')) {
            $this->autoValidate = $this->container->call([$this, 'autoValidate']);
        }

        if ($this->autoValidate) {
            $this->handleValidate();
        }
    }

    /**
     * Handle validate.
     */
    protected function handleValidate()
    {
        parent::validateResolved();
    }

    /**
     * 定义 FormRequest->getValidatorInstance 下 validator 验证器
     *
     * @param Factory $factory
     *
     * @return Validator
     */
    public function validator(Factory $factory)
    {
        $validationData = $this->isMethod('GET') ? $this->query() : $this->post();
        return $factory->make($validationData, $this->getRules(), $this->messages(), $this->attributes());
    }

    /**
     * Get rules.
     *
     * @return array
     */
    protected function getRules()
    {
        return $this->handleScene($this->container->call([$this, 'rules']));
    }

    /**
     * Handle scene.
     *
     * @param array $rules
     *
     * @return array
     */
    protected function handleScene(array $rules)
    {
        if ($this->onlyRule) {
            return $this->handleRules($this->onlyRule, $rules);
        }

        if (!empty($this->scene) && method_exists($this, 'scene')) {
            $scene = $this->container->call([$this, 'scene']);
            if (array_key_exists($this->scene, $scene)) {
                return $this->handleRules($scene[$this->scene], $rules);
            }
        }

        return $rules;
    }

    /**
     * Handle rules.
     *
     * @param array $sceneRules
     * @param array $rules
     *
     * @return array
     */
    protected function handleRules(array $sceneRules, array $rules)
    {
        $result = [];
        foreach ($sceneRules as $key => $value) {
            if (is_numeric($key) && array_key_exists($value, $rules)) {
                $result[$value] = $rules[$value];
            } else {
                $result[$key] = $value;
            }
        }

        return $result;
    }
}

新建「请求类」

  • 新建「请求类」 app/Http/Requests/PermissionRequest.php
<?php
    
namespace App\Http\Requests;

use Illuminate\Validation\Rule;

class PermissionRequest extends FormRequest
{
    protected $autoValidate = false;

    /**
     * Get the validation rules that apply to the request.
     *
     * @return array
     */
    public function rules()
    {
        return [
            //全部的验证规则
            'name' => ['required', 'string', 'between:2,16'],
            'type' => ['required', 'int', Rule::in([1, 2])],
            'user_ids' => ['required', 'array'],
            'role_id' => ['required', 'integer'],
            'user_id' => ['required', 'integer'],
            'action_codes' => ['required', 'array'],
        ];
    }

    /**
     * Get custom attributes for validator errors.
     *
     * @return array
     */
    public function attributes()
    {
        return [
            'name' => '角色名称',
            'type' => '角色类型',
            'user_ids' => '用户ID集',
            'role_id' => '角色ID',
            'user_id' => '用户ID',
            'action_codes' => '行为编码集',
        ];
    }

    /**
     * scene.
     *
     * @return array
     */
    public function scene()
    {
        return [
            'roleCreate' => ['name', 'type'],
            'roleMemberAdd' => ['user_ids', 'role_id'],
            'roleMemberDel' => ['role_id', 'user_id'],
            'roleMemberList' => ['role_id'],
            'roleActionEdit' => ['role_id', 'action_codes'],
            'roleActionList' => ['role_id'],
            'allActionMarkList' => ['role_id'],
        ];
    }
}

编辑异常处理文件

  • 编辑 app/Exception/Handler.php

  • Laravel 6.*

public function report(Exception $exception)
{
  parent::report($exception);
}

public function render($request, Exception $exception)
{
  $commonHandler = new CommonHandler();
  // 验证错误
  if ($request->is("api/*")) {// 我们仅仅处理 api的异常，所以这里要排除下其他前缀路由
    //如果抛出的异常是 DefineValidationException 的实例，我们就可以确定该异常是表单验证异常
    if ($exception instanceof \Illuminate\Validation\ValidationException) {
      // 下面是你需要包装的数据
      $message = $exception->validator->errors()->first();// 更好的获取错误的方法
      return $this->authMessage(400, $message);
    }

    //处理请求方法异常
    if ($exception instanceof MethodNotAllowedHttpException){
      return $this->authMessage(400, $$exception->getMessage());
    }


    //处理内部错误异常
    if (($exception instanceof FatalThrowableError || $exception instanceof \ErrorException || $exception instanceof QueryException) && env('APP_ENV') != 'local') {
      return $this->authMessage(400, $$exception->getMessage());
    }
  }

  return parent::render($request, $exception);
}
  • Laravel 8.*
public function register()
{
  $this->reportable(function (Throwable $e) {
    //
  });

  //捕捉表单验证错误
  $this->renderable(function (ValidationException $exception, $request) {
    $message = $exception->validator->errors()->first();// 更好的获取错误的方法
    return response()->json(['status' => 40000, 'msg' => $message, 'data' => []]);
  });
}
