Laravel中发送Ajax请求🏃‍♂️

ajax请求验证,添加数据

$("#submit2").click(function () {
    var _error = 0;
    if ($(".addtit").val() == "") {
        _error = 1
    }
    if (_error == 1) {
        layer.msg('所有信息必须填写', {icon: 5, time: 980});
        return false;
    }
    $.ajax({
        type:'post',
        url:'{{route('messages.sysstore')}}',
        data:{
            '_token':{{csrf_filed()}},
            'type':1,
            'title':$('#title_2').val(),
            'timing':$("#timing_input2").val(),
            'images': JSON.stringify( images_arr )
        },
        success:function(msg){
            if(msg.code == 0){
                layer.msg('ok！',{icon:6,time:1500},function(){
                    location.href = "{{route('messages.index')}}";
                });
            }else{
                layer.msg(msg.error,{icon:5,time:1500});
            }
        }
    });
    return false;
})

$.ajax({
     url: '/registerAgreement/' + _id,
     type: 'PATCH',
      success: function (data) {
      console.log(data);
      if (data[0] == 1) {
            var _img = data[1] ? 'watch1' : 'watch01';
            _this.find(".watch").attr("src", "/assets/images/" + _img + ".png");
            return layer.msg("启用成功！", {icon: 6, time: 1500},function () {

         })

           return layer.alert("启用失败！", {icon: 5, time: 3000})
